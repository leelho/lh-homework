### 1）用伪代码描述将十进制转换成16进制  

    ```c

     #include<stdio.h>
     void f(int*,int*);

     int main ()
     {
	
	 int a,b;
	int *i;
	int *p;
	a=100;
	b=200;
	i=&a;
	p=&b;
	f(i,p);
	

	
	printf("%d %d",a,b);
    }
    void f(int *a,int *b)
    {
    int temp;
    temp=*a;
    *a=*b;
    *b=temp;

    }
    ```
